import { cache } from "react";

// React.cache() deduplicates within a single server request
// (generateMetadata + page component both call this — only one DB query)
export const cachedGetProductBySlug = cache(getProductBySlug);

**Part A: Parallelize checkout DB queries in `actions/checkout.ts`**

In `createCheckoutSession()`, the products query, variants query, and settings query are currently sequential. Refactor to run them in parallel:

```typescript
// Before (sequential):
const { data: dbProducts } = await supabase.from("products")...
const { data: dbVariants } = variantIds.length ? await supabase.from("product_variants")... : { data: [] };
// ... later ...
const settings = await getShopSettings();

// After (parallel):
const [productsResult, variantsResult, settings] = await Promise.all([
  supabase.from("products").select("id, price, stock_quantity, is_active, name").in("id", productIds),
  variantIds.length
    ? supabase.from("product_variants").select("id, price, stock_quantity, is_active, product_id").in("id", variantIds)
    : Promise.resolve({ data: [] }),
  getShopSettings(),
]);

const dbProducts = productsResult.data;
const dbVariants = variantsResult.data;
```

The coupon query MUST remain sequential after this block because it depends on the subtotal calculated from product prices.

**Part B: Parallelize variant Stripe updates in `actions/admin-products.ts`**

In `updateProduct()`, the `for...of` loop on variants (around lines 401-474) processes each variant sequentially. Refactor to use `Promise.all()`:

Replace the `for (const v of variants)` loop with:
```typescript
await Promise.all(variants.map(async (v) => {
  if (v.id) {
    // Update existing variant
    const prev = existingVariantMap.get(v.id);
    let variantStripePriceId = prev?.stripePriceId ?? null;

    if (stripeProductId && Math.round(v.price * 100) !== Math.round((prev?.price ?? 0) * 100)) {
      const variantPrice = await stripe.prices.create({
        product: stripeProductId,
        unit_amount: Math.round(v.price * 100),
        currency: CURRENCY,
        metadata: { variant_name: v.name },
      });
      if (prev?.stripePriceId) {
        await stripe.prices.update(prev.stripePriceId, { active: false });
      }
      variantStripePriceId = variantPrice.id;
    }

    await supabase.from("product_variants").update({
      name: v.name, sku: v.sku, price: v.price,
      compare_at_price: v.compare_at_price,
      stock_quantity: v.stock_quantity, sort_order: v.sort_order,
      is_active: v.is_active, stripe_price_id: variantStripePriceId,
      variant_type: v.variant_type, color_hex: v.color_hex,
      swatch_image: v.swatch_image, variant_images: v.variant_images,
      description: v.description, updated_at: new Date().toISOString(),
    }).eq("id", v.id);
  } else {
    // Insert new variant
    let variantStripePriceId: string | null = null;
    if (stripeProductId) {
      const variantPrice = await stripe.prices.create({
        product: stripeProductId,
        unit_amount: Math.round(v.price * 100),
        currency: CURRENCY,
        metadata: { variant_name: v.name },
      });
      variantStripePriceId = variantPrice.id;
    }
    await supabase.from("product_variants").insert({
      product_id: id, name: v.name, sku: v.sku, price: v.price,
      compare_at_price: v.compare_at_price,
      stock_quantity: v.stock_quantity, sort_order: v.sort_order,
      is_active: v.is_active, stripe_price_id: variantStripePriceId,
      variant_type: v.variant_type, color_hex: v.color_hex,
      swatch_image: v.swatch_image, variant_images: v.variant_images,
      description: v.description,
    });
  }
}));
```

NOTE: The deletion of removed variants (lines 394-399) can stay before this Promise.all since it's a single batch operation.
