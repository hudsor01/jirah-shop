import { z } from "zod";

// Shared fragments reused across action files
export const emailSchema = z.string().email("Please enter a valid email address");

export const passwordSchema = z.string()
  .min(8, "Password must be at least 8 characters")
  .regex(/[A-Z]/, "Password must contain at least 1 uppercase letter")
  .regex(/[0-9]/, "Password must contain at least 1 number");

export const uuidSchema = z.string().uuid("Invalid ID format");

export const paginationSchema = z.object({
  page: z.number().int().positive().optional(),
  limit: z.number().int().min(1).max(100).optional(),
});

// Helper to extract first user-readable error message from ZodError
export function formatZodError(error: z.ZodError): string {
  return error.issues[0]?.message ?? "Invalid input";
}
```

This file provides the shared fragments that CONTEXT.md specifies: emailSchema shared across auth + contact, passwordSchema for registration, uuidSchema for ID validation, paginationSchema for list endpoints.

The `formatZodError` helper provides a consistent way to extract user-readable messages without exposing Zod internals.

const ProductFormDataSchema = z.object({
  name: z.string().min(1, "Product name is required").max(200),
  slug: z.string().min(1, "Slug is required").max(200),
  description: z.string(),
  short_description: z.string().max(500),
  price: z.number().positive("Price must be positive"),
  compare_at_price: z.number().positive().nullable(),
  category: z.string().min(1, "Category is required"),
  subcategory: z.string().nullable(),
  brand: z.string().min(1, "Brand is required"),
  is_own_brand: z.boolean(),
  images: z.array(z.string().url()).max(8, "Maximum 8 images"),
  ingredients: z.string().nullable(),
  how_to_use: z.string().nullable(),
  tags: z.array(z.string()),
  stock_quantity: z.number().int().min(0, "Stock cannot be negative"),
  has_variants: z.boolean(),
  is_featured: z.boolean(),
  is_active: z.boolean(),
});

const VariantFormDataSchema = z.object({
  id: z.string().optional(),
  name: z.string().min(1, "Variant name is required"),
  sku: z.string().min(1, "SKU is required"),
  price: z.number().positive("Price must be positive"),
  compare_at_price: z.number().positive().nullable(),
  stock_quantity: z.number().int().min(0, "Stock cannot be negative"),
  sort_order: z.number().int().min(0),
  is_active: z.boolean(),
  variant_type: z.string().min(1),
  color_hex: z.string().nullable(),
  swatch_image: z.string().nullable(),
  variant_images: z.array(z.string()).nullable(),
  description: z.string().nullable(),
});

const AdminProductsOptionsSchema = paginationSchema.extend({
  search: z.string().optional(),
  category: z.string().optional(),
});
```

Add safeParse() validation to each function:

1. **getAdminProducts**: Validate `options` with `AdminProductsOptionsSchema.safeParse(options ?? {})`. On failure, return `{ products: [], count: 0 }`.

2. **getAdminProduct**: Validate `id` with `uuidSchema.safeParse(id)`. On failure, return `null`.

3. **createProduct**: Validate `formData` with `ProductFormDataSchema.safeParse(formData)` and `variants` with `z.array(VariantFormDataSchema).safeParse(variants)`. On failure, return `{ success: false, error: formatZodError(...) }`.

4. **updateProduct**: Validate `id` with `uuidSchema.safeParse(id)`, `formData` with `ProductFormDataSchema.safeParse(formData)`, and `variants` with `z.array(VariantFormDataSchema).safeParse(variants)`. On failure, return `{ success: false, error: formatZodError(...) }`.

5. **deleteProduct**: Validate `id` with `uuidSchema.safeParse(id)`. On failure, return `{ success: false, error: "Invalid product ID" }`.

Place all validation AFTER `requireAdmin()` but BEFORE any database or Stripe calls. The existing TypeScript `ProductFormData` and `VariantFormData` types remain (used by callers); the Zod schemas validate at runtime.
