<interfaces>
<!-- Query function pattern: pure data access, throws on error -->
```typescript
// queries/products.ts
import { createClient } from "@/lib/supabase/server";

export async function queryProducts(options?: {
  category?: string;
  search?: string;
  sort?: string;
  limit?: number;
  page?: number;
}): Promise<{ data: Product[]; count: number }> {
  const supabase = await createClient();
  // Pure Supabase query
  const { data, error, count } = await supabase.from("products")...;
  if (error) throw error;
  return { data: data ?? [], count: count ?? 0 };
}
```

<!-- Action function pattern: delegates to query, wraps in ActionResult -->
```typescript
// actions/products.ts
import { queryProducts } from "@/queries/products";
import { ok, fail } from "@/lib/action-result";

export async function getProducts(options?: {...}): Promise<ActionResult<...>> {
  try {
    const { data, count } = await queryProducts(options);
    return ok({ data, total: count, page: options?.page ?? 1, pageSize: options?.limit ?? 20 });
  } catch (e) {
    return fail(e instanceof Error ? e.message : "Failed to fetch products");
  }
}
```

<!-- Cached query pattern: wraps query function directly -->
```typescript
// lib/cached-queries.ts
import { queryProducts } from "@/queries/products";

export async function cachedGetProducts(options?: {...}) {
  "use cache";
  cacheTag("products");
  return queryProducts(options);
}
```
