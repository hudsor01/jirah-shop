  **CQ-04: Shared checkout hook**

  1. Create `hooks/use-checkout.ts`:
     ```typescript
     "use client";
     import { useState } from "react";
     import { useCart } from "@/providers/cart-provider";
     import { createCheckoutSession } from "@/actions/checkout";
     import { validateCartPrices } from "@/actions/products";
     import { toast } from "sonner";

     export function useCheckout() {
       const { items, updateItemPrices } = useCart();
       const [isLoading, setIsLoading] = useState(false);

       async function checkout(couponCode?: string) {
         setIsLoading(true);
         try {
           // Validate prices
           const priceResult = await validateCartPrices(
             items.map(item => ({ id: item.id, variantId: item.variantId, expectedPrice: item.price }))
           );
           if (priceResult.success && !priceResult.data.valid && priceResult.data.updates) {
             updateItemPrices(priceResult.data.updates);
             toast.error("Some prices have changed. Please review your cart.");
             return;
           }

           // Create checkout session
           const sessionResult = await createCheckoutSession(items, couponCode);
           if (sessionResult.success && sessionResult.data.url) {
             window.location.href = sessionResult.data.url;
           } else if (!sessionResult.success) {
             toast.error(sessionResult.error);
           }
         } catch {
           toast.error("Checkout failed. Please try again.");
         } finally {
           setIsLoading(false);
         }
       }

       return { checkout, isLoading };
     }
     ```
     Note: Adapt the exact parameter shapes to match what validateCartPrices and createCheckoutSession expect after Plan 01 conversion.
