```typescript
import { test, expect } from '@playwright/test';

test.describe('Smoke Test', () => {
  test('homepage loads and displays products', async ({ page }) => {
    await page.goto('/');

    // Verify page loads
    await expect(page).toHaveTitle(/jirah/i);

    // Verify products section exists and has items
    // Use a flexible selector — look for product cards or product grid
    const productSection = page.locator('[data-testid="products"]').or(
      page.locator('main')
    );
    await expect(productSection).toBeVisible();
  });

  test('can navigate to a product page', async ({ page }) => {
    await page.goto('/');

    // Find and click the first product link
    const firstProduct = page.locator('a[href^="/products/"]').first();
    // Only run this if products exist on page
    const count = await firstProduct.count();
    if (count > 0) {
      await firstProduct.click();
      // Verify we're on a product page
      await expect(page.url()).toContain('/products/');
      // Product page should have an add to cart button
      const addToCartButton = page.getByRole('button', { name: /add to cart/i });
      await expect(addToCartButton).toBeVisible();
    }
  });

  test('can add item to cart and see cart drawer', async ({ page }) => {
    await page.goto('/');

    // Find the first add to cart button on homepage (product card)
    const addToCartBtn = page.getByRole('button', { name: /add to cart/i }).first();
    const btnCount = await addToCartBtn.count();

    if (btnCount > 0) {
      await addToCartBtn.click();

      // Verify cart indicator updates (cart icon shows count)
      // or cart drawer/sheet opens
      // Look for cart drawer or toast notification
      await expect(
        page.getByText(/added to cart/i).or(page.locator('[data-testid="cart-drawer"]'))
      ).toBeVisible({ timeout: 5000 });
    }
  });
});
```

Important notes for the executor:
- The test should be RESILIENT — use `.or()` selectors and conditional logic so it doesn't break on minor UI changes
- Don't assume specific product data exists (the test env may have empty DB)
- Use data-testid attributes where they exist, fall back to role/text selectors
- The tests should pass even if the dev server has no products (skip gracefully)
- Do NOT navigate to /checkout or interact with Stripe
